[docs]def setlocale(loc=None):
    
    """
      
      
    **setlocale**

    Function for setting locale


    """

    if loc is None:
        l = locale.getdefaultlocale()[0]
    else:
        l = loc
    lc = gettext.translation('main', localedir='locales', languages=[l])
    lc.install()
    return lc.gettext, lc.ngettext
