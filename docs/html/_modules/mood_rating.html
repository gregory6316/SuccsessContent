def setlocale(loc=None):
    if loc is None:
        l = locale.getdefaultlocale()[0]
    else:
        l = loc
    lc = gettext.translation('mood_rating', localedir='locales', languages=[l])
    lc.install()
    return lc.gettext, lc.ngettext

        """
